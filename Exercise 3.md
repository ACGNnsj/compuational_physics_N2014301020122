data=[[0 for i in range(80)] for raw in range(80)]

    data[0][i]=1
    data[1][i]=1
for i in range(80):
    data[39][i] = 1
    data[40][i] = 1
    data[i][39] = 1
    data[i][40] = 1

#flag = '1'
while 1:
    a=len(data)
    for i in range(a):
        for j in range(0,a):
            if data[i][j] == 1:
                print '*',
            else:
                print' ',
        print '\n',

    print '\n'
    a=len(data)
    b = [[0 for i in range(80)] for raw in range(80)]
    for i in range(a):
        for j in range(a):
            b[a - 1 - j][a - 1 - i] = data[i][a - 1 - j]
    data = b
